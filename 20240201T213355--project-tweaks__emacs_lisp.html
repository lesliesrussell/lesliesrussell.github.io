;;; -*- lexical-binding: t; -*-

(setq project-switch-commands
      '((project-find-file "Find file")
        (project-find-regexp "Find regexp")
        (project-find-dir "Find directory")
        (project-vc-dir "VC-Dir")
        (project-dired "Dired")
        (project-eshell "Eshell")))

(defun g-project-lgrep ()
  "Search in the current project using lgrep."
  (interactive)
  (let ((project-root default-directory)
        (search-term (read-string "Search for: ")))
    (lgrep search-term "*.*" project-root)))

(use-package project
  :bind (:map project-prefix-map
              ("s" . 'g-project-lgrep)))

(defun g-eshell ()
  "Start eshell at the root of the current project, or in the
current directory if the current buffer is not part of a
project."
  (interactive)
  (if (project-current)
      (project-eshell)
    (eshell)))

(global-set-key (kbd "C-x s") 'g-eshell)
