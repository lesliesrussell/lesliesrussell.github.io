(require 'pulse)

(defun pulse-line (&rest _)
  "Pulse current line."
  (pulse-momentary-highlight-one-line (point)))

(dolist (command '(scroll-up-command scroll-down-dommand
                                     recenter-top-bottom other-window))
  (advice-add command :after #'pulse-line))
