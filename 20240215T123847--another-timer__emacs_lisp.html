(defvar my-timer nil "Variable to store the active timer.")

(defun start-my-timer ()
  "Starts a timer and stores it in `my-timer'."
  (interactive)
  (setq my-timer (run-with-timer 5 nil (lambda () (message "Timer ended after 5 seconds!"))))
  (message "Timer started!"))

(defun stop-my-timer ()
  "Stops the timer stored in `my-timer'."
  (interactive)
  (when (timerp my-timer)
    (cancel-timer my-timer)
    (setq my-timer nil)
    (message "Timer stopped manually.")))

(global-set-key (kbd "<f5>") 'start-my-timer)
(global-set-key (kbd "<f6>") 'stop-my-timer)
