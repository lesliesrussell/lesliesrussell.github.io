(defun eww-view-page-with-pup ()
  (interactive)
  (let ((url (eww-current-url)))
    (unless url
      (error "No URL to process"))
    (let ((temp-file (make-temp-file "eww-pup-" nil ".html")))
      (with-temp-buffer
        (shell-command (format "/home/lsr/.local/bin/readable '%s'" url) t)
        (write-region (point-min) (point-max) temp-file))
      (eww-open-file temp-file))))

(with-eval-after-load 'eww
   ;; This ensures `eww-mode-map` is defined
   (define-key eww-mode-map (kbd "C-c e") 'eww-view-page-with-pup))
